\frametitle{MuonHIP CRAFT program}
\begin{itemize}
\item Large structure pre-alignment: wheel and disk positions (6 dof) relative to tracker
\item Alignment of 200 out of 250 DT chambers relative to tracker \mbox{(fewer dof)\hspace{-1 cm}}
\item Two-stage alignment of CSCs: overlaps procedure for
  chambers relative to rings, followed by ring alignment (similar to
  \mbox{disk alignment)\hspace{-1 cm}}
\end{itemize}
